## 快速开始

### 1. 基础设置

```typescript
// main.ts - 游戏主入口
import { World } from './core/World';
import { MovementSystem } from './systems/MovementSystem';
import { RenderSystem } from './systems/RenderSystem';
import { HealthSystem } from './systems/HealthSystem';
import { PositionComponent, VelocityComponent, HealthComponent, SpriteComponent } from './components';

// 创建世界实例
const world = new World();

// 注册组件类型

world.registerComponentType('Velocity', VelocityComponent);
world.registerComponentType('Health', HealthComponent);
world.registerComponentType('Sprite', SpriteComponent);

// 添加系统
world.addSystem(new MovementSystem(world.componentManager, world.entityFactory, world.eventBus));
world.addSystem(new RenderSystem(world.componentManager, world.entityFactory, world.eventBus));
world.addSystem(new HealthSystem(world.componentManager, world.entityFactory, world.eventBus));

// 启动世界
world.start();

// 游戏主循环
function gameLoop(): void {
  const deltaTime = 16; // 60 FPS
  world.update(deltaTime);
  
  // 继续下一帧
  setTimeout(gameLoop, deltaTime);
}

gameLoop();
```

```typescript
// systems/MovementSystem.ts
import { System } from '../core/System';
import { PositionComponent, VelocityComponent } from '../components';

export class MovementSystem extends System {
  private readonly REQUIRED_COMPONENTS = ['Position', 'Velocity'];
  
  update(deltaTime: number): void {
    if (!this.isEnabled) return;
    
    const entities = this.componentManager.queryEntities(this.REQUIRED_COMPONENTS);
    
    for (const entityId of entities) {
      const entity = { id: entityId, version: 1 };
      
      const position = this.componentManager.getComponent<PositionComponent>(entity, 'Position');
      const velocity = this.componentManager.getComponent<VelocityComponent>(entity, 'Velocity');
      
      if (position && velocity) {
        // 应用速度限制
        const speed = Math.sqrt(velocity.x * velocity.x + velocity.y * velocity.y);
        if (speed > velocity.maxSpeed) {
          const factor = velocity.maxSpeed / speed;
          velocity.x *= factor;
          velocity.y *= factor;
        }
        
        // 更新位置
        position.x += velocity.x * deltaTime / 1000;
        position.y += velocity.y * deltaTime / 1000;
        position.z += velocity.z * deltaTime / 1000;
      }
    }
  }
}
```

### 示例3：AI行为系统

```typescript
// AI相关组件
export class AIComponent implements Component {
  readonly __componentType = 'AI';
  
  constructor(
    public behaviorType: 'aggressive' | 'defensive' | 'patrol' = 'aggressive',
    public detectionRange: number = 100,
    public target?: EntityId
  ) {}
}

export class PatrolComponent implements Component {
  readonly __componentType = 'Patrol';
  
  constructor(
    public points: Array<{x: number, y: number}> = [],
    public currentIndex: number = 0,
    public speed: number = 20
  ) {}
}

// AI系统
export class AISystem extends System {
  private readonly REQUIRED_COMPONENTS = ['AI', 'Position'];
  
  update(deltaTime: number): void {
    if (!this.isEnabled) return;
    
    const aiEntities = this.componentManager.queryEntities(this.REQUIRED_COMPONENTS);
    
    for (const entityId of aiEntities) {
      const entity = { id: entityId, version: 1 };
      const ai = this.componentManager.getComponent<AIComponent>(entity, 'AI');
      
      if (ai) {
        switch (ai.behaviorType) {
          case 'aggressive':
            this.processAggressiveBehavior(entity, ai);
            break;
          case 'defensive':
            this.processDefensiveBehavior(entity, ai);
            break;
          case 'patrol':
            this.processPatrolBehavior(entity);
            break;
        }
      }
    }
  }
  
  private processAggressiveBehavior(entity: Entity, ai: AIComponent): void {
    const position = this.componentManager.getComponent<PositionComponent>(entity, 'Position');
    if (!position) return;
    
    // 寻找最近的敌人
    const target = this.findNearestEnemy(entity, position, ai.detectionRange);
    
    if (target) {
      ai.target = target.id;
      this.moveTowardsTarget(entity, target);
    } else {
      ai.target = undefined;
      this.stopMovement(entity);
    }
  }
  
  private processPatrolBehavior(entity: Entity): void {
    const patrol = this.componentManager.getComponent<PatrolComponent>(entity, 'Patrol');
    const position = this.componentManager.getComponent<PositionComponent>(entity, 'Position');
    const velocity = this.componentManager.getComponent<VelocityComponent>(entity, 'Velocity');
    
    if (!patrol || !position || !velocity || patrol.points.length === 0) return;
    
    const targetPoint = patrol.points[patrol.currentIndex];
    const distance = Math.sqrt(
      Math.pow(targetPoint.x - position.x, 2) + 
      Math.pow(targetPoint.y - position.y, 2)
    );
    
    if (distance < 10) {
      // 到达巡逻点，移动到下一个
      patrol.currentIndex = (patrol.currentIndex + 1) % patrol.points.length;
    } else {
      // 移动向目标点
      const direction = {
        x: (targetPoint.x - position.x) / distance,
        y: (targetPoint.y - position.y) / distance
      };
      
      velocity.x = direction.x * patrol.speed;
      velocity.y = direction.y * patrol.speed;
    }
  }
  
  private findNearestEnemy(entity: Entity, position: PositionComponent, range: number): Entity | undefined {
    // 查找范围内的敌人
    const potentialTargets = this.componentManager.queryEntities(['Position', 'Health']);
    let nearestEnemy: Entity | undefined;
    let nearestDistance = range;
    
    for (const targetId of potentialTargets) {
      if (targetId === entity.id) continue; // 跳过自己
      
      const targetEntity = { id: targetId, version: 1 };
      const targetPosition = this.componentManager.getComponent<PositionComponent>(targetEntity, 'Position');
      
      if (targetPosition) {
        const distance = Math.sqrt(
          Math.pow(targetPosition.x - position.x, 2) + 
          Math.pow(targetPosition.y - position.y, 2)
        );
        
        if (distance < nearestDistance) {
          nearestDistance = distance;
          nearestEnemy = targetEntity;
        }
      }
    }
    
    return nearestEnemy;
  }
  
  private moveTowardsTarget(entity: Entity, target: Entity): void {
    const position = this.componentManager.getComponent<PositionComponent>(entity, 'Position');
    const velocity = this.componentManager.getComponent<VelocityComponent>(entity, 'Velocity');
    const targetPosition = this.componentManager.getComponent<PositionComponent>(target, 'Position');
    
    if (position && velocity && targetPosition) {
      const distance = Math.sqrt(
        Math.pow(targetPosition.x - position.x, 2) + 
        Math.pow(targetPosition.y - position.y, 2)
      );
      
      if (distance > 0) {
        const direction = {
          x: (targetPosition.x - position.x) / distance,
          y: (targetPosition.y - position.y) / distance
        };
        
        velocity.x = direction.x * velocity.maxSpeed;
        velocity.y = direction.y * velocity.maxSpeed;
      }
    }
  }
  
  private stopMovement(entity: Entity): void {
    this.componentManager.updateComponent<VelocityComponent>(entity, 'Velocity', (velocity) => {
      velocity.x = 0;
      velocity.y = 0;
      velocity.z = 0;
    });
  }
}
```

```typescript
// 技能相关组件
export class SkillComponent implements Component {
  readonly __componentType = 'Skill';
  
  constructor(
    public skills: Map<string, Skill> = new Map(),
    public cooldowns: Map<string, number> = new Map()
  ) {}
}

export interface Skill {
  id: string;
  name: string;
  cooldown: number;
  manaCost: number;
  range: number;
  effect: SkillEffect;
}

export interface SkillEffect {
  type: 'damage' | 'heal' | 'buff' | 'debuff';
  value: number;
  duration?: number;
  area?: number;
}

export class ManaComponent implements Component {
  readonly __componentType = 'Mana';
  
  constructor(
    public current: number = 100,
    public maximum: number = 100,
    public regeneration: number = 1
  ) {}
}

// 技能系统
export class SkillSystem extends System {
  initialize(): void {
    this.eventBus.subscribe('CastSkill', (event: CastSkillEvent) => {
      this.castSkill(event.casterId, event.skillId, event.targetId, event.targetPosition);
    });
  }
  
  update(deltaTime: number): void {
    this.updateCooldowns(deltaTime);
    this.updateManaRegeneration(deltaTime);
  }
  
  private castSkill(
    casterId: EntityId, 
    skillId: string, 
    targetId?: EntityId, 
    targetPosition?: {x: number, y: number}
  ): void {
    const caster = { id: casterId, version: 1 };
    const skillComponent = this.componentManager.getComponent<SkillComponent>(caster, 'Skill');
    const mana = this.componentManager.getComponent<ManaComponent>(caster, 'Mana');
    
    if (!skillComponent || !mana) return;
    
    const skill = skillComponent.skills.get(skillId);
    if (!skill) return;
    
    // 检查冷却时间
    const currentCooldown = skillComponent.cooldowns.get(skillId) || 0;
    if (currentCooldown > 0) return;
    
    // 检查法力值
    if (mana.current < skill.manaCost) return;
    
    // 检查范围（如果有目标）
    if (targetId && !this.isInRange(caster, { id: targetId, version: 1 }, skill.range)) {
      return;
    }
    
    // 消耗法力值
    mana.current -= skill.manaCost;
    
    // 设置冷却时间
    skillComponent.cooldowns.set(skillId, skill.cooldown);
    
    // 执行技能效果
    this.executeSkillEffect(caster, skill, targetId, targetPosition);
    
    // 发布技能释放事件
    this.eventBus.publish({
      type: 'SkillCast',
      casterId,
      skillId,
      targetId,
      targetPosition,
      timestamp: Date.now()
    });
  }
  
  private executeSkillEffect(
    caster: Entity, 
    skill: Skill, 
    targetId?: EntityId, 
    targetPosition?: {x: number, y: number}
  ): void {
    switch (skill.effect.type) {
      case 'damage':
        if (targetId) {
          this.eventBus.publish({
            type: 'DamageDealt',
            attackerId: caster.id,
            targetId,
            damage: skill.effect.value,
            timestamp: Date.now()
          });
        }
        break;
        
      case 'heal':
        if (targetId) {
          const target = { id: targetId, version: 1 };
          this.componentManager.updateComponent<HealthComponent>(target, 'Health', (health) => {
            health.current = Math.min(health.maximum, health.current + skill.effect.value);
          });
        }
        break;
        
      case 'buff':
      case 'debuff':
        // 实现增益/减益效果
        this.applyStatusEffect(targetId || caster.id, skill.effect);
        break;
    }
  }
  
  private updateCooldowns(deltaTime: number): void {
    const entities = this.componentManager.queryEntities(['Skill']);
    
    for (const entityId of entities) {
      const entity = { id: entityId, version: 1 };
      const skillComponent = this.componentManager.getComponent<SkillComponent>(entity, 'Skill');
      
      if (skillComponent) {
        for (const [skillId, cooldown] of skillComponent.cooldowns) {
          const newCooldown = Math.max(0, cooldown - deltaTime);
          if (newCooldown === 0) {
            skillComponent.cooldowns.delete(skillId);
          } else {
            skillComponent.cooldowns.set(skillId, newCooldown);
          }
        }
      }
    }
  }
  
  private updateManaRegeneration(deltaTime: number): void {
    const entities = this.componentManager.queryEntities(['Mana']);
    
    for (const entityId of entities) {
      const entity = { id: entityId, version: 1 };
      this.componentManager.updateComponent<ManaComponent>(entity, 'Mana', (mana) => {
        mana.current = Math.min(mana.maximum, mana.current + mana.regeneration * deltaTime / 1000);
      });
    }
  }
  
  private isInRange(caster: Entity, target: Entity, range: number): boolean {
    const casterPos = this.componentManager.getComponent<PositionComponent>(caster, 'Position');
    const targetPos = this.componentManager.getComponent<PositionComponent>(target, 'Position');
    
    if (!casterPos || !targetPos) return false;

      Math.pow(targetPos.x - casterPos.x, 2) + 
      Math.pow(targetPos.y - casterPos.y, 2)
    );
    
    return distance <= range;
  }

}
```

```typescript
  constructor(createFn: () => T, resetFn: (obj: T) => void, initialSize: number = 10) {
    this.createFn = createFn;
    this.resetFn = resetFn;
    
    // 预创建对象
    for (let i = 0; i < initialSize; i++) {
      this.pool.push(this.createFn());
    }
  }
  
  get(): T {
    if (this.pool.length > 0) {
      return this.pool.pop()!;
    }
    return this.createFn();
  }
  
  release(obj: T): void {
    this.resetFn(obj);
    this.pool.push(obj);
  }
}

// 使用对象池的子弹系统
class BulletSystem extends System {
  private bulletPool: ObjectPool<Entity>;
  
  constructor(componentManager: ComponentManager, entityFactory: EntityFactory, eventBus: EventBus) {
    super(componentManager, entityFactory, eventBus);
    
    this.bulletPool = new ObjectPool(
      () => this.createBullet(),
      (bullet) => this.resetBullet(bullet),
      50 // 预创建50个子弹
    );
  }
  
  fireBullet(shooterId: EntityId, targetX: number, targetY: number): void {
    const bullet = this.bulletPool.get();
    const shooterPos = this.componentManager.getComponent<PositionComponent>(
      { id: shooterId, version: 1 }, 
      'Position'
    );
    
    if (shooterPos) {
      // 设置子弹位置和方向
      this.componentManager.updateComponent<PositionComponent>(bullet, 'Position', (pos) => {
        pos.x = shooterPos.x;
        pos.y = shooterPos.y;
      });
      
      const direction = this.calculateDirection(shooterPos, { x: targetX, y: targetY });
      this.componentManager.updateComponent<VelocityComponent>(bullet, 'Velocity', (vel) => {
        vel.x = direction.x * 200; // 子弹速度
        vel.y = direction.y * 200;
      });
      
      // 激活子弹
      this.componentManager.updateComponent<SpriteComponent>(bullet, 'Sprite', (sprite) => {
        sprite.visible = true;
      });
    }
  }
  
  private createBullet(): Entity {
    const bullet = this.entityFactory.create();
    
    this.componentManager.addComponent(bullet, new PositionComponent());
    this.componentManager.addComponent(bullet, new VelocityComponent());
    this.componentManager.addComponent(bullet, new SpriteComponent('bullet.png', 8, 8, false));
    this.componentManager.addComponent(bullet, new DamageComponent(10));
    
    return bullet;
  }
  
  private resetBullet(bullet: Entity): void {
    // 重置子弹状态
    this.componentManager.updateComponent<SpriteComponent>(bullet, 'Sprite', (sprite) => {
      sprite.visible = false;
    });
    
    this.componentManager.updateComponent<VelocityComponent>(bullet, 'Velocity', (vel) => {
      vel.x = 0;
      vel.y = 0;
    });
  }
}
```
