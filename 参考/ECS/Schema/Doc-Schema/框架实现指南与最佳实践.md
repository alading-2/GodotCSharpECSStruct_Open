```typescript
// core/EntityFactory.ts
export class EntityFactory {
  private nextId: EntityId = 1;
  private recycledIds: EntityId[] = [];
  private entityVersions = new Map<EntityId, number>();
  private maxRecycledIds = 1000; // 限制回收ID数量

  create(): Entity {
    const id = this.getNextId();
    const version = this.getNextVersion(id);
    
    return { id, version };
  }

  destroy(entity: Entity): boolean {
    if (!this.isValid(entity)) {
      return false;
    }

    // 回收ID（如果回收池未满）
    if (this.recycledIds.length < this.maxRecycledIds) {
      this.recycledIds.push(entity.id);
    }

    return true;
  }

  isValid(entity: Entity): boolean {
    const currentVersion = this.entityVersions.get(entity.id);
    return currentVersion !== undefined && currentVersion === entity.version;
  }

  // 批量创建实体
  createBatch(count: number): Entity[] {
    const entities: Entity[] = [];
    for (let i = 0; i < count; i++) {
      entities.push(this.create());
    }
    return entities;
  }

  // 获取统计信息
  getStats(): { totalCreated: number; recycledCount: number; activeCount: number } {
    return {
      totalCreated: this.nextId - 1,
      recycledCount: this.recycledIds.length,
      activeCount: this.entityVersions.size - this.recycledIds.length
    };
  }

  private getNextId(): EntityId {
    return this.recycledIds.pop() || this.nextId++;
  }

  private getNextVersion(id: EntityId): number {
    const currentVersion = this.entityVersions.get(id) || 0;
    const newVersion = currentVersion + 1;
    this.entityVersions.set(id, newVersion);
    return newVersion;
  }

  // 清理过期版本信息
  cleanup(): void {
    // 清理不再使用的实体版本信息
    const activeIds = new Set<EntityId>();
    
    // 这里需要从ComponentManager获取活跃的实体ID
    // 为简化，这里只是示例
    for (const [id, version] of this.entityVersions) {
      if (!activeIds.has(id)) {
        this.entityVersions.delete(id);
      }
    }
  }
}
```

### 3. 高性能 ComponentArray 实现

```typescript
// core/ComponentArray.ts
export class ComponentArray<T extends Component> {
  private components: T[] = [];
  private entityToIndex = new Map<EntityId, number>();
  private indexToEntity = new Map<number, EntityId>();
  private freeIndices: number[] = [];
  private componentType: string;

  constructor(componentType: string) {
    this.componentType = componentType;
  }

  add(entity: Entity, component: T): void {
    if (this.entityToIndex.has(entity.id)) {
      throw new Error(`Entity ${entity.id} already has component ${this.componentType}`);
    }

    const index = this.allocateIndex();
    this.components[index] = component;
    this.entityToIndex.set(entity.id, index);
    this.indexToEntity.set(index, entity.id);
  }

  remove(entity: Entity): T | undefined {
    const index = this.entityToIndex.get(entity.id);
    if (index === undefined) return undefined;

    const component = this.components[index];
    this.deallocateIndex(index, entity.id);
    
    return component;
  }

  get(entity: Entity): T | undefined {
    const index = this.entityToIndex.get(entity.id);
    return index !== undefined ? this.components[index] : undefined;
  }

  has(entity: Entity): boolean {
    return this.entityToIndex.has(entity.id);
  }

  update(entity: Entity, updater: (component: T) => void): boolean {
    const component = this.get(entity);
    if (component) {
      updater(component);
      return true;
    }
    return false;
  }

  // 批量操作
  getAllComponents(): readonly T[] {
    return this.components.filter((_, index) => this.indexToEntity.has(index));
  }

  getEntitiesWithComponent(): EntityId[] {
    return Array.from(this.entityToIndex.keys());
  }

  // 迭代器支持
  *[Symbol.iterator](): Iterator<[EntityId, T]> {
    for (const [entityId, index] of this.entityToIndex) {
      yield [entityId, this.components[index]];
    }
  }

  // 性能优化方法
  compact(): void {
    const newComponents: T[] = [];
    const newEntityToIndex = new Map<EntityId, number>();
    const newIndexToEntity = new Map<number, EntityId>();

    let newIndex = 0;
    for (const [entityId, oldIndex] of this.entityToIndex) {
      newComponents[newIndex] = this.components[oldIndex];
      newEntityToIndex.set(entityId, newIndex);
      newIndexToEntity.set(newIndex, entityId);
      newIndex++;
    }

    this.components = newComponents;
    this.entityToIndex = newEntityToIndex;
    this.indexToEntity = newIndexToEntity;
    this.freeIndices = [];
  }

  // 内存使用统计
  getMemoryStats(): { totalSlots: number; usedSlots: number; fragmentation: number } {
    const totalSlots = this.components.length;
    const usedSlots = this.entityToIndex.size;
    const fragmentation = totalSlots > 0 ? (totalSlots - usedSlots) / totalSlots : 0;
    
    return { totalSlots, usedSlots, fragmentation };
  }

  private allocateIndex(): number {
    return this.freeIndices.pop() ?? this.components.length;
  }

  private deallocateIndex(index: number, entityId: EntityId): void {
    const lastIndex = this.components.length - 1;
    
    if (index === lastIndex) {
      // 移除最后一个元素
      this.components.pop();
    } else {
      // 移动最后一个元素到当前位置
      const lastEntityId = this.indexToEntity.get(lastIndex)!;
      const lastComponent = this.components[lastIndex];
      
      this.components[index] = lastComponent;
      this.entityToIndex.set(lastEntityId, index);
      this.indexToEntity.set(index, lastEntityId);
      
      this.components.pop();
      this.indexToEntity.delete(lastIndex);
    }
    
    this.entityToIndex.delete(entityId);
  }
}
```

### 4. 智能 ComponentManager 实现

```typescript
// core/ComponentManager.ts
export class ComponentManager {
  private componentArrays = new Map<string, ComponentArray<any>>();
  private entitySignatures = new Map<EntityId, ComponentSignature>();
  private typeRegistry = new Map<string, ComponentConstructor>();
  private eventBus?: EventBus;
  private queryCache = new QueryCache();

  constructor(eventBus?: EventBus) {
    this.eventBus = eventBus;
  }

  // 注册组件类型
  registerComponentType<T extends Component>(
    typeName: string, 
    constructor: ComponentConstructor<T>
  ): void {
    if (this.componentArrays.has(typeName)) {
      throw new Error(`Component type '${typeName}' already registered`);
    }

    this.componentArrays.set(typeName, new ComponentArray<T>(typeName));
    this.typeRegistry.set(typeName, constructor);
  }

  // 创建组件实例
  createComponent<T extends Component>(typeName: string, data?: Partial<T>): T {
    const constructor = this.typeRegistry.get(typeName);
    if (!constructor) {
      throw new Error(`Component type '${typeName}' not registered`);
    }
    return new constructor(data) as T;
  }

  // 添加组件
  addComponent<T extends Component>(entity: Entity, component: T): void {
    const typeName = component.__componentType;
    
    if (!this.componentArrays.has(typeName)) {
      throw new Error(`Component type '${typeName}' not registered`);
    }

    const array = this.componentArrays.get(typeName)!;
    array.add(entity, component);

    // 更新实体签名
    this.updateEntitySignature(entity.id, typeName, true);
    
    // 发布事件
    this.eventBus?.publish({
      type: 'ComponentAdded',
      entityId: entity.id,
      componentType: typeName,
      timestamp: Date.now()
    });

    // 使查询缓存失效
    this.queryCache.invalidateQuery(typeName);
  }

  // 移除组件
  removeComponent<T extends Component>(entity: Entity, typeName: string): T | undefined {
    const array = this.componentArrays.get(typeName);
    if (!array) return undefined;

    const component = array.remove(entity);
    if (component) {
      this.updateEntitySignature(entity.id, typeName, false);
      
      this.eventBus?.publish({
        type: 'ComponentRemoved',
        entityId: entity.id,
        componentType: typeName,
        timestamp: Date.now()
      });

      this.queryCache.invalidateQuery(typeName);
    }

    return component;
  }

  // 获取组件
  getComponent<T extends Component>(entity: Entity, typeName: string): T | undefined {
    const array = this.componentArrays.get(typeName);
    return array?.get(entity);
  }

  // 更新组件
  updateComponent<T extends Component>(
    entity: Entity, 
    typeName: string, 
    updater: (component: T) => void
  ): boolean {
    const array = this.componentArrays.get(typeName);
    return array?.update(entity, updater) ?? false;
  }

  // 检查组件
  hasComponent(entity: Entity, typeName: string): boolean {
    return this.entitySignatures.get(entity.id)?.has(typeName) ?? false;
  }

  // 检查多个组件
  hasComponents(entity: Entity, componentTypes: string[]): boolean {
    const signature = this.entitySignatures.get(entity.id);
    if (!signature) return false;
    
    return componentTypes.every(type => signature.has(type));
  }

  // 获取实体签名
  getEntitySignature(entity: Entity): ComponentSignature {
    return this.entitySignatures.get(entity.id) ?? new Set();
  }

  // 高效查询实体
  queryEntities(componentTypes: string[]): EntityId[] {
    // 尝试从缓存获取
    const cached = this.queryCache.getQuery(componentTypes);
    if (cached) {
      return cached;
    }

    // 执行查询
    const result = this.executeQuery(componentTypes);
    
    // 缓存结果
    this.queryCache.setQuery(componentTypes, result);
    
    return result;
  }

  // 查询单个组件类型的所有实体
  queryEntitiesWith(componentType: string): EntityId[] {
    const array = this.componentArrays.get(componentType);
    return array?.getEntitiesWithComponent() ?? [];
  }

  // 批量获取组件
  getComponentsOfType<T extends Component>(typeName: string): readonly T[] {
    const array = this.componentArrays.get(typeName);
    return array?.getAllComponents() ?? [];
  }

  // 销毁实体的所有组件
  destroyEntity(entity: Entity): void {
    const signature = this.entitySignatures.get(entity.id);
    if (!signature) return;

    // 移除所有组件
    for (const typeName of signature) {
      this.removeComponent(entity, typeName);
    }

    this.entitySignatures.delete(entity.id);
  }

  // 获取统计信息
  getStats(): ComponentManagerStats {
    const stats: ComponentManagerStats = {
      totalComponentTypes: this.componentArrays.size,
      totalEntities: this.entitySignatures.size,
      componentCounts: new Map(),
      memoryUsage: new Map()
    };

    for (const [typeName, array] of this.componentArrays) {
      const entities = array.getEntitiesWithComponent();
      const memStats = array.getMemoryStats();
      
      stats.componentCounts.set(typeName, entities.length);
      stats.memoryUsage.set(typeName, memStats);
    }

    return stats;
  }

  // 内存优化
  compact(): void {
    for (const array of this.componentArrays.values()) {
      array.compact();
    }
  }

  private executeQuery(componentTypes: string[]): EntityId[] {
    if (componentTypes.length === 0) return [];
    
    // 优化：从最小的组件集合开始
    const sortedTypes = [...componentTypes].sort((a, b) => {
      const aCount = this.componentArrays.get(a)?.getEntitiesWithComponent().length ?? 0;
      const bCount = this.componentArrays.get(b)?.getEntitiesWithComponent().length ?? 0;
      return aCount - bCount;
    });

    let candidates = this.queryEntitiesWith(sortedTypes[0]);
    
    for (let i = 1; i < sortedTypes.length && candidates.length > 0; i++) {
      const signature = new Set(this.queryEntitiesWith(sortedTypes[i]));
      candidates = candidates.filter(entityId => signature.has(entityId));
    }

    return candidates;
  }

  private updateEntitySignature(entityId: EntityId, componentType: string, add: boolean): void {
    if (!this.entitySignatures.has(entityId)) {
      this.entitySignatures.set(entityId, new Set());
    }

    const signature = this.entitySignatures.get(entityId)!;
    if (add) {
      signature.add(componentType);
    } else {
      signature.delete(componentType);
      
      // 如果实体没有任何组件，移除签名
      if (signature.size === 0) {
        this.entitySignatures.delete(entityId);
      }
    }
  }
}

// 统计信息接口
interface ComponentManagerStats {
  totalComponentTypes: number;
  totalEntities: number;
  componentCounts: Map<string, number>;
  memoryUsage: Map<string, { totalSlots: number; usedSlots: number; fragmentation: number }>;
}
```

### 5. 查询缓存优化

```typescript
// core/QueryCache.ts
export class QueryCache {
  private cache = new Map<string, QueryResult>();
  private invalidationSet = new Set<string>();
  private maxCacheSize = 100;
  private cacheHits = 0;
  private cacheMisses = 0;

  getQuery(componentTypes: string[]): EntityId[] | undefined {
    const key = this.getQueryKey(componentTypes);
    
    if (this.invalidationSet.has(key)) {
      this.cache.delete(key);
      this.invalidationSet.delete(key);
      this.cacheMisses++;
      return undefined;
    }
    
    const result = this.cache.get(key);
    if (result) {
      this.cacheHits++;
      return result.entities;
    }
    
    this.cacheMisses++;
    return undefined;
  }

  setQuery(componentTypes: string[], entities: EntityId[]): void {
    const key = this.getQueryKey(componentTypes);
    
    // 限制缓存大小
    if (this.cache.size >= this.maxCacheSize) {
      this.evictOldestEntry();
    }
    
    this.cache.set(key, {
      entities: [...entities],
      timestamp: Date.now()
    });
  }

  invalidateQuery(componentType: string): void {
    // 标记所有包含该组件类型的查询为无效
    for (const key of this.cache.keys()) {
      if (this.keyContainsComponent(key, componentType)) {
        this.invalidationSet.add(key);
      }
    }
  }

  clear(): void {
    this.cache.clear();
    this.invalidationSet.clear();
  }

  getStats(): { hitRate: number; totalQueries: number; cacheSize: number } {
    const totalQueries = this.cacheHits + this.cacheMisses;
    return {
      hitRate: totalQueries > 0 ? this.cacheHits / totalQueries : 0,
      totalQueries,
      cacheSize: this.cache.size
    };
  }

  private getQueryKey(componentTypes: string[]): string {
    return componentTypes.sort().join('|');
  }

  private keyContainsComponent(key: string, componentType: string): boolean {
    return key.split('|').includes(componentType);
  }

  private evictOldestEntry(): void {
    let oldestKey: string | undefined;
    let oldestTimestamp = Infinity;
    
    for (const [key, result] of this.cache) {
      if (result.timestamp < oldestTimestamp) {
        oldestTimestamp = result.timestamp;
        oldestKey = key;
      }
    }
    
    if (oldestKey) {
      this.cache.delete(oldestKey);
    }
  }
}
```

```typescript
// core/EventBus.ts
export interface GameEvent {
  readonly type: string;
  readonly timestamp: number;
}

export interface ComponentAddedEvent extends GameEvent {
  type: 'ComponentAdded';
  entityId: EntityId;
  componentType: string;
}

export interface ComponentRemovedEvent extends GameEvent {
  type: 'ComponentRemoved';
  entityId: EntityId;
  componentType: string;
}

export interface EntityCreatedEvent extends GameEvent {
  type: 'EntityCreated';
  entityId: EntityId;
}

export interface EntityDestroyedEvent extends GameEvent {
  type: 'EntityDestroyed';
  entityId: EntityId;
}

export type EventListener<T extends GameEvent = GameEvent> = (event: T) => void;

export class EventBus {
  private listeners = new Map<string, Set<EventListener>>();
  private eventQueue: GameEvent[] = [];
  private isProcessing = false;
  private maxQueueSize = 1000;

  subscribe<T extends GameEvent>(eventType: string, listener: EventListener<T>): () => void {
    if (!this.listeners.has(eventType)) {
      this.listeners.set(eventType, new Set());
    }
    
    this.listeners.get(eventType)!.add(listener as EventListener);
    
    // 返回取消订阅函数
    return () => {
      const listeners = this.listeners.get(eventType);
      if (listeners) {
        listeners.delete(listener as EventListener);
        if (listeners.size === 0) {
          this.listeners.delete(eventType);
        }
      }
    };
  }

  publish(event: GameEvent): void {
    if (this.eventQueue.length >= this.maxQueueSize) {
      console.warn('Event queue overflow, dropping oldest events');
      this.eventQueue.shift();
    }
    
    this.eventQueue.push(event);
    
    if (!this.isProcessing) {
      this.processEvents();
    }
  }

  publishImmediate(event: GameEvent): void {
    const listeners = this.listeners.get(event.type);
    if (listeners) {
      for (const listener of listeners) {
        try {
          listener(event);
        } catch (error) {
          console.error(`Error in event listener for ${event.type}:`, error);
        }
      }
    }
  }

  private processEvents(): void {
    this.isProcessing = true;
    
    while (this.eventQueue.length > 0) {
      const event = this.eventQueue.shift()!;
      this.publishImmediate(event);
    }
    
    this.isProcessing = false;
  }

  clear(): void {
    this.listeners.clear();
    this.eventQueue = [];
  }

  getStats(): { listenerCount: number; queueSize: number } {
    let totalListeners = 0;
    for (const listeners of this.listeners.values()) {
      totalListeners += listeners.size;
    }
    
    return {
      listenerCount: totalListeners,
      queueSize: this.eventQueue.length
    };
  }
}
```

```typescript
// core/System.ts
export abstract class System {
  protected componentManager: ComponentManager;
  protected entityFactory: EntityFactory;
  protected eventBus: EventBus;
  protected isEnabled = true;
  protected priority = 0;

  constructor(
    componentManager: ComponentManager,
    entityFactory: EntityFactory,
    eventBus: EventBus
  ) {
    this.componentManager = componentManager;
    this.entityFactory = entityFactory;
    this.eventBus = eventBus;
  }

  abstract update(deltaTime: number): void;
  
  initialize?(): void;
  cleanup?(): void;
  
  enable(): void {
    this.isEnabled = true;
  }
  
  disable(): void {
    this.isEnabled = false;
  }
  
  setPriority(priority: number): void {
    this.priority = priority;
  }
  
  getPriority(): number {
    return this.priority;
  }
}

// 具体系统示例
export class MovementSystem extends System {
  private readonly REQUIRED_COMPONENTS = ['Position', 'Velocity'];
  
  update(deltaTime: number): void {
    if (!this.isEnabled) return;
    
    const entities = this.componentManager.queryEntities(this.REQUIRED_COMPONENTS);
    
    for (const entityId of entities) {
      const entity = { id: entityId, version: 1 }; // 简化版本检查
      
      this.componentManager.updateComponent<PositionComponent>(
        entity, 
        'Position', 
        (position) => {
          const velocity = this.componentManager.getComponent<VelocityComponent>(entity, 'Velocity');
          if (velocity) {
            position.x += velocity.x * deltaTime;
            position.y += velocity.y * deltaTime;
            position.z += velocity.z * deltaTime;
          }
        }
      );
    }
  }
}

export class HealthSystem extends System {
  initialize(): void {
    // 监听伤害事件
    this.eventBus.subscribe('DamageDealt', (event: DamageEvent) => {
      this.applyDamage(event.targetEntityId, event.damage);
    });
  }
  
  update(deltaTime: number): void {
    if (!this.isEnabled) return;
    
    const entities = this.componentManager.queryEntities(['Health']);
    
    for (const entityId of entities) {
      const entity = { id: entityId, version: 1 };
      const health = this.componentManager.getComponent<HealthComponent>(entity, 'Health');
      
      if (health && health.current <= 0) {
        this.handleEntityDeath(entity);
      }
    }
  }
  
  private applyDamage(entityId: EntityId, damage: number): void {
    const entity = { id: entityId, version: 1 };
    
    this.componentManager.updateComponent<HealthComponent>(
      entity,
      'Health',
      (health) => {
        health.current = Math.max(0, health.current - damage);
      }
    );
  }
  
  private handleEntityDeath(entity: Entity): void {
    this.eventBus.publish({
      type: 'EntityDied',
      entityId: entity.id,
      timestamp: Date.now()
    });
    
    this.componentManager.destroyEntity(entity);
    this.entityFactory.destroy(entity);
  }
}
```

```typescript
// core/World.ts
export class World {
  private componentManager: ComponentManager;
  private entityFactory: EntityFactory;
  private eventBus: EventBus;
  private systems: System[] = [];
  private isRunning = false;
  private lastUpdateTime = 0;
  private frameCount = 0;
  private performanceMonitor = new PerformanceMonitor();

  constructor() {
    this.eventBus = new EventBus();
    this.componentManager = new ComponentManager(this.eventBus);
    this.entityFactory = new EntityFactory();
  }

  // 系统管理
  addSystem(system: System): void {
    this.systems.push(system);
    this.systems.sort((a, b) => a.getPriority() - b.getPriority());
    
    system.initialize?.();
  }

  removeSystem(system: System): void {
    const index = this.systems.indexOf(system);
    if (index !== -1) {
      system.cleanup?.();
      this.systems.splice(index, 1);
    }

    return this.systems.find(system => system instanceof systemClass) as T;

    const entity = this.entityFactory.create();
    
    this.eventBus.publish({
      type: 'EntityCreated',
      entityId: entity.id,

    });
    
    return entity;

    if (!this.entityFactory.isValid(entity)) {
      return false;

    this.componentManager.destroyEntity(entity);
    const success = this.entityFactory.destroy(entity);
    
    if (success) {
      this.eventBus.publish({
        type: 'EntityDestroyed',
        entityId: entity.id,
        timestamp: Date.now()

    updater: (component: T) => void
  ): boolean {
    return this.componentManager.updateComponent(entity, typeName, updater);
  }

  // 查询
  queryEntities(componentTypes: string[]): EntityId[] {
    return this.componentManager.queryEntities(componentTypes);
  }

  // 事件系统
  subscribe<T extends GameEvent>(eventType: string, listener: EventListener<T>): () => void {
    return this.eventBus.subscribe(eventType, listener);
  }

  publish(event: GameEvent): void {
    this.eventBus.publish(event);
  }

  // 世界更新
  update(deltaTime: number): void {
    if (!this.isRunning) return;
    
    const startTime = performance.now();
    
    // 更新所有系统
    for (const system of this.systems) {
      const systemStartTime = performance.now();
      system.update(deltaTime);
      const systemEndTime = performance.now();
      
      this.performanceMonitor.recordSystemTime(
        system.constructor.name,
        systemEndTime - systemStartTime
      );
    }
    
    const endTime = performance.now();
    this.performanceMonitor.recordFrameTime(endTime - startTime);
    
    this.frameCount++;
    this.lastUpdateTime = Date.now();
  }

  // 世界控制
  start(): void {
    this.isRunning = true;
    this.lastUpdateTime = Date.now();
  }

  stop(): void {
    this.isRunning = false;
  }

  pause(): void {
    this.isRunning = false;
  }

  resume(): void {
    this.isRunning = true;
    this.lastUpdateTime = Date.now();
  }

  // 统计和调试
  getStats(): WorldStats {
    const entityStats = this.entityFactory.getStats();
    const componentStats = this.componentManager.getStats();
    const eventStats = this.eventBus.getStats();
    const performanceStats = this.performanceMonitor.getStats();
    
    return {
      frameCount: this.frameCount,
      isRunning: this.isRunning,
      systemCount: this.systems.length,
      entityStats,
      componentStats,
      eventStats,
      performanceStats
    };
  }

  // 内存优化
  optimize(): void {
    this.componentManager.compact();
    this.entityFactory.cleanup();
  }

  // 清理
  cleanup(): void {
    this.stop();
    
    for (const system of this.systems) {
      system.cleanup?.();
    }
    
    this.systems = [];
    this.eventBus.clear();
    this.frameCount = 0;
  }
}
```

```typescript
// utils/PerformanceMonitor.ts
export class PerformanceMonitor {
  private systemTimes = new Map<string, number[]>();
  private frameTimes: number[] = [];
  private maxSamples = 100;

  recordSystemTime(systemName: string, time: number): void {
    if (!this.systemTimes.has(systemName)) {
      this.systemTimes.set(systemName, []);
    }
    
    const times = this.systemTimes.get(systemName)!;
    times.push(time);
    
    if (times.length > this.maxSamples) {
      times.shift();
    }
  }

  recordFrameTime(time: number): void {
    this.frameTimes.push(time);
    
    if (this.frameTimes.length > this.maxSamples) {
      this.frameTimes.shift();
    }
  }

  getStats(): PerformanceStats {
    const systemStats = new Map<string, SystemPerformanceStats>();
    
    for (const [systemName, times] of this.systemTimes) {
      if (times.length > 0) {
        const avg = times.reduce((sum, time) => sum + time, 0) / times.length;
        const min = Math.min(...times);
        const max = Math.max(...times);
        
        systemStats.set(systemName, { avg, min, max, sampleCount: times.length });
      }
    }
    
    const frameStats = this.calculateFrameStats();
    
    return {
      systemStats,
      frameStats
    };
  }

  private calculateFrameStats(): FramePerformanceStats {
    if (this.frameTimes.length === 0) {
      return { avg: 0, min: 0, max: 0, fps: 0, sampleCount: 0 };
    }
    
    const avg = this.frameTimes.reduce((sum, time) => sum + time, 0) / this.frameTimes.length;
    const min = Math.min(...this.frameTimes);
    const max = Math.max(...this.frameTimes);
    const fps = avg > 0 ? 1000 / avg : 0;
    
    return {
      avg,
      min,
      max,
      fps,
      sampleCount: this.frameTimes.length
    };
  }

  clear(): void {
    this.systemTimes.clear();
    this.frameTimes = [];
  }
}

interface SystemPerformanceStats {
  avg: number;
  min: number;
  max: number;
  sampleCount: number;
}

interface FramePerformanceStats {
  avg: number;
  min: number;
  max: number;
  fps: number;
  sampleCount: number;
}

interface PerformanceStats {
  systemStats: Map<string, SystemPerformanceStats>;
  frameStats: FramePerformanceStats;
}
```

```typescript
  }
```

```typescript
class PerformanceProfiler {
  private profiles = new Map<string, PerformanceProfile>();
  
  startProfile(name: string): void {
    this.profiles.set(name, {
      startTime: performance.now(),
      endTime: 0,
      duration: 0
    });
  }
  
  endProfile(name: string): number {
    const profile = this.profiles.get(name);
    if (profile) {
      profile.endTime = performance.now();
      profile.duration = profile.endTime - profile.startTime;
      return profile.duration;
    }
    return 0;
  }
  
  getReport(): PerformanceReport {
    const report: PerformanceReport = {
      profiles: Array.from(this.profiles.entries()).map(([name, profile]) => ({
        name,
        duration: profile.duration
      })).sort((a, b) => b.duration - a.duration)
    };
    
    return report;
  }
}

interface PerformanceProfile {
  startTime: number;
  endTime: number;
  duration: number;
}

  profiles: Array<{ name: string; duration: number }>;
```
