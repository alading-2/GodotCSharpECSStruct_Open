# 现代数据管理框架设计方案

## 框架概述

基于对Unity ECS、Unreal Engine、Supernova Engine等现代游戏引擎的深度分析，设计一个全新的、独立的数据管理框架。该框架严格遵循现代游戏设计原则，实现数据与逻辑的完全分离，提供高性能、类型安全的数据管理能力。

## 核心设计原则

### 1. 数据导向设计 (Data-Oriented Design)
- **纯数据组件**：组件只包含数据，零逻辑
- **系统处理逻辑**：所有业务逻辑在系统中处理
- **内存局部性**：数据按类型连续存储

### 2. 严格的职责分离
- **Entity**：纯粹的ID容器，无任何逻辑
- **Component**：纯数据结构，无方法
- **System**：纯逻辑处理，通过依赖注入访问数据
- **Manager**：数据管理和生命周期控制

### 3. 现代架构模式
- **依赖注入**：避免全局单例
- **工厂模式**：统一创建和管理
- **观察者模式**：事件驱动的数据变更
- **策略模式**：可配置的验证和序列化

## 框架架构设计

### 1. 核心架构层次

```
┌─────────────────────────────────────┐
│           System Layer              │  ← 业务逻辑系统
│  (MovementSystem, CombatSystem...)  │
├─────────────────────────────────────┤
│          Manager Layer              │  ← 数据管理层
│   (EntityManager, ComponentManager) │
├─────────────────────────────────────┤
│          Storage Layer              │  ← 数据存储层
│    (ComponentArray, DataStore)     │
├─────────────────────────────────────┤
│          Entity Layer               │  ← 实体标识层
│        (Entity, EntityId)           │
└─────────────────────────────────────┘
```

### 2. 数据流向设计

```
System → Manager → Storage → Component Data
  ↓        ↓         ↓           ↓
逻辑处理  类型管理   索引映射    原始数据
```

```typescript
// Entity: 纯粹的ID概念
type EntityId = number;

interface Entity {
  readonly id: EntityId;
  readonly version: number; // 用于检测Entity是否已被销毁
}

class EntityFactory {
  private nextId: EntityId = 1;
  private recycledIds: EntityId[] = [];
  private entityVersions = new Map<EntityId, number>();

  create(): Entity {
    const id = this.recycledIds.pop() || this.nextId++;
    const version = (this.entityVersions.get(id) || 0) + 1;
    this.entityVersions.set(id, version);
    
    return { id, version };
  }

  destroy(entity: Entity): void {
    if (this.isValid(entity)) {
      this.recycledIds.push(entity.id);
    }
  }

  isValid(entity: Entity): boolean {
    return this.entityVersions.get(entity.id) === entity.version;
  }
}
```

```typescript
// 组件类型注册
class ComponentTypeRegistry {
  private static typeMap = new Map<string, ComponentConstructor>();
  private static nextTypeId = 0;
  private static typeIds = new Map<string, number>();

  static register<T extends Component>(typeName: string, constructor: ComponentConstructor<T>): void {
    this.typeMap.set(typeName, constructor);
    this.typeIds.set(typeName, this.nextTypeId++);
  }

  static getTypeId(typeName: string): number {
    const id = this.typeIds.get(typeName);
    if (id === undefined) {
      throw new Error(`Component type '${typeName}' not registered`);
    }
    return id;
  }

  static create<T extends Component>(typeName: string, data?: Partial<T>): T {
    const constructor = this.typeMap.get(typeName);
    if (!constructor) {
      throw new Error(`Component type '${typeName}' not registered`);
    }
    return new constructor(data) as T;
  }
}
```

### 3. ComponentArray 存储设计

```typescript
// 高性能组件数组存储
class ComponentArray<T extends Component> {
  private components: T[] = [];
  private entityToIndex = new Map<EntityId, number>();
  private indexToEntity = new Map<number, EntityId>();
  private freeIndices: number[] = [];

  add(entity: Entity, component: T): void {
    if (this.entityToIndex.has(entity.id)) {
      throw new Error(`Entity ${entity.id} already has component`);
    }

    const index = this.freeIndices.pop() ?? this.components.length;
    
    if (index === this.components.length) {
      this.components.push(component);
    } else {
      this.components[index] = component;
    }

    this.entityToIndex.set(entity.id, index);
    this.indexToEntity.set(index, entity.id);
  }

  remove(entity: Entity): T | undefined {
    const index = this.entityToIndex.get(entity.id);
    if (index === undefined) return undefined;

    const component = this.components[index];
    
    // 移动最后一个元素到当前位置（保持数组紧凑）
    const lastIndex = this.components.length - 1;
    if (index !== lastIndex) {
      const lastComponent = this.components[lastIndex];
      const lastEntityId = this.indexToEntity.get(lastIndex)!;
      
      this.components[index] = lastComponent;
      this.entityToIndex.set(lastEntityId, index);
      this.indexToEntity.set(index, lastEntityId);
    }

    this.components.pop();
    this.entityToIndex.delete(entity.id);
    this.indexToEntity.delete(lastIndex);
    
    if (index !== lastIndex) {
      this.freeIndices.push(lastIndex);
    }

    return component;
  }

  get(entity: Entity): T | undefined {
    const index = this.entityToIndex.get(entity.id);
    return index !== undefined ? this.components[index] : undefined;
  }

  has(entity: Entity): boolean {
    return this.entityToIndex.has(entity.id);
  }

  // 批量操作支持
  getAllComponents(): readonly T[] {
    return this.components;
  }

  getEntitiesWithComponent(): EntityId[] {
    return Array.from(this.entityToIndex.keys());
  }

  // 内存优化
  compact(): void {
    // 重新整理数组，移除空隙
    const newComponents: T[] = [];
    const newEntityToIndex = new Map<EntityId, number>();
    const newIndexToEntity = new Map<number, EntityId>();

    let newIndex = 0;
    for (const [entityId, oldIndex] of this.entityToIndex) {
      newComponents[newIndex] = this.components[oldIndex];
      newEntityToIndex.set(entityId, newIndex);
      newIndexToEntity.set(newIndex, entityId);
      newIndex++;
    }

    this.components = newComponents;
    this.entityToIndex = newEntityToIndex;
    this.indexToEntity = newIndexToEntity;
    this.freeIndices = [];
  }
}
```

### 4. ComponentManager 设计

```typescript
// 组件管理器
class ComponentManager {
  private componentArrays = new Map<string, ComponentArray<any>>();
  private entitySignatures = new Map<EntityId, Set<string>>();

  // 注册组件类型
  registerComponentType<T extends Component>(typeName: string): void {
    if (!this.componentArrays.has(typeName)) {
      this.componentArrays.set(typeName, new ComponentArray<T>());
    }
  }

  // 添加组件
  addComponent<T extends Component>(entity: Entity, component: T): void {
    const typeName = component.__componentType;
    
    if (!this.componentArrays.has(typeName)) {
      this.registerComponentType<T>(typeName);
    }

    const array = this.componentArrays.get(typeName)!;
    array.add(entity, component);

    // 更新实体签名
    if (!this.entitySignatures.has(entity.id)) {
      this.entitySignatures.set(entity.id, new Set());
    }
    this.entitySignatures.get(entity.id)!.add(typeName);
  }

  // 移除组件
  removeComponent<T extends Component>(entity: Entity, typeName: string): T | undefined {
    const array = this.componentArrays.get(typeName);
    if (!array) return undefined;

    const component = array.remove(entity);
    if (component) {
      this.entitySignatures.get(entity.id)?.delete(typeName);
    }

    return component;
  }

  // 获取组件
  getComponent<T extends Component>(entity: Entity, typeName: string): T | undefined {
    const array = this.componentArrays.get(typeName);
    return array?.get(entity);
  }

  // 检查组件
  hasComponent(entity: Entity, typeName: string): boolean {
    return this.entitySignatures.get(entity.id)?.has(typeName) ?? false;
  }

  // 获取实体的所有组件类型
  getEntitySignature(entity: Entity): Set<string> {
    return this.entitySignatures.get(entity.id) ?? new Set();
  }

  // 查询具有特定组件的实体
  queryEntities(componentTypes: string[]): EntityId[] {
    const result: EntityId[] = [];
    
    for (const [entityId, signature] of this.entitySignatures) {
      if (componentTypes.every(type => signature.has(type))) {
        result.push(entityId);
      }
    }
    
    return result;
  }

  // 批量获取组件
  getComponentsOfType<T extends Component>(typeName: string): readonly T[] {
    const array = this.componentArrays.get(typeName);
    return array?.getAllComponents() ?? [];
  }

  // 销毁实体的所有组件
  destroyEntity(entity: Entity): void {
    const signature = this.entitySignatures.get(entity.id);
    if (!signature) return;

    for (const typeName of signature) {
      this.removeComponent(entity, typeName);
    }

    this.entitySignatures.delete(entity.id);
  }
}
```

### 5. System 设计

```typescript
// 系统基类
abstract class System {
  protected componentManager: ComponentManager;
  protected entityFactory: EntityFactory;

  constructor(componentManager: ComponentManager, entityFactory: EntityFactory) {
    this.componentManager = componentManager;
    this.entityFactory = entityFactory;
  }

  abstract update(deltaTime: number): void;
  
  // 可选的初始化和清理方法
  initialize?(): void;
  cleanup?(): void;
}

// 具体系统示例
class MovementSystem extends System {
  update(deltaTime: number): void {
    // 查询所有具有Position和Velocity组件的实体
    const entities = this.componentManager.queryEntities(['Position', 'Velocity']);
    
    for (const entityId of entities) {
      const entity = { id: entityId, version: 1 }; // 简化版本检查
      const position = this.componentManager.getComponent<PositionComponent>(entity, 'Position');
      const velocity = this.componentManager.getComponent<VelocityComponent>(entity, 'Velocity');
      
      if (position && velocity) {
        position.x += velocity.x * deltaTime;
        position.y += velocity.y * deltaTime;
        position.z += velocity.z * deltaTime;
      }
    }
  }
}

class HealthSystem extends System {
  update(deltaTime: number): void {
    const entities = this.componentManager.queryEntities(['Health']);
    
    for (const entityId of entities) {
      const entity = { id: entityId, version: 1 };
      const health = this.componentManager.getComponent<HealthComponent>(entity, 'Health');
      
      if (health && health.current <= 0) {
        // 实体死亡，销毁实体
        this.componentManager.destroyEntity(entity);
        this.entityFactory.destroy(entity);
      }
    }
  }
}
```

```typescript
// 世界容器，管理所有系统和组件
class World {
  private componentManager: ComponentManager;
  private entityFactory: EntityFactory;
  private systems: System[] = [];
  private isRunning = false;

  constructor() {
    this.componentManager = new ComponentManager();
    this.entityFactory = new EntityFactory();
  }

  // 注册系统
  addSystem(system: System): void {
    this.systems.push(system);
    system.initialize?.();
  }

  // 移除系统
  removeSystem(system: System): void {
    const index = this.systems.indexOf(system);
    if (index !== -1) {
      system.cleanup?.();
      this.systems.splice(index, 1);
    }
  }

  // 创建实体
  createEntity(): Entity {
    return this.entityFactory.create();
  }

  // 销毁实体
  destroyEntity(entity: Entity): void {
    this.componentManager.destroyEntity(entity);
    this.entityFactory.destroy(entity);
  }

  // 添加组件到实体
  addComponent<T extends Component>(entity: Entity, component: T): void {
    this.componentManager.addComponent(entity, component);
  }

  // 从实体移除组件
  removeComponent<T extends Component>(entity: Entity, typeName: string): T | undefined {
    return this.componentManager.removeComponent<T>(entity, typeName);
  }

  // 获取实体组件
  getComponent<T extends Component>(entity: Entity, typeName: string): T | undefined {
    return this.componentManager.getComponent<T>(entity, typeName);
  }

  // 检查实体是否有组件
  hasComponent(entity: Entity, typeName: string): boolean {
    return this.componentManager.hasComponent(entity, typeName);
```

```typescript
interface GameEvent {
  readonly type: string;
  readonly timestamp: number;
}

class EventBus {
  private listeners = new Map<string, Array<(event: GameEvent) => void>>();

  subscribe(eventType: string, listener: (event: GameEvent) => void): () => void {
    if (!this.listeners.has(eventType)) {
      this.listeners.set(eventType, []);
    }
    
    this.listeners.get(eventType)!.push(listener);
    
    // 返回取消订阅函数
    return () => {
      const listeners = this.listeners.get(eventType);
      if (listeners) {
        const index = listeners.indexOf(listener);
        if (index !== -1) {
          listeners.splice(index, 1);
        }
      }
    };
  }

  publish(event: GameEvent): void {
    const listeners = this.listeners.get(event.type);
    if (listeners) {
      for (const listener of listeners) {
        listener(event);
      }
    }
  }
}
```

```typescript
// 查询缓存
class QueryCache {
  private cache = new Map<string, EntityId[]>();
  private isDirty = new Set<string>();

  getQuery(componentTypes: string[]): EntityId[] | undefined {
    const key = this.getQueryKey(componentTypes);
    
    if (this.isDirty.has(key)) {
      this.cache.delete(key);
      this.isDirty.delete(key);
      return undefined;
    }
    
    return this.cache.get(key);
  }

  setQuery(componentTypes: string[], entities: EntityId[]): void {
    const key = this.getQueryKey(componentTypes);
    this.cache.set(key, [...entities]);
  }

  invalidateQuery(componentType: string): void {
    // 标记所有包含该组件类型的查询为脏
    for (const key of this.cache.keys()) {
      if (key.includes(componentType)) {
        this.isDirty.add(key);
      }
    }
  }

  private getQueryKey(componentTypes: string[]): string {
    return componentTypes.sort().join('|');
  }
}
```

## 性能优化策略

### 1. 内存池

```typescript
// 对象池
class ObjectPool<T> {
  private pool: T[] = [];
  private createFn: () => T;
  private resetFn: (obj: T) => void;

  constructor(createFn: () => T, resetFn: (obj: T) => void, initialSize = 10) {
    this.createFn = createFn;
    this.resetFn = resetFn;
    
    // 预分配对象
    for (let i = 0; i < initialSize; i++) {
      this.pool.push(createFn());
    }
  }

  acquire(): T {
    return this.pool.pop() || this.createFn();
  }

  release(obj: T): void {
    this.resetFn(obj);
    this.pool.push(obj);
  }
}
```

### 2. 批量操作

```typescript
// 批量组件操作
class BatchComponentManager {
  private pendingOperations: Array<() => void> = [];
  private batchTimer: number | null = null;

  addComponentBatch<T extends Component>(entity: Entity, component: T): void {
    this.pendingOperations.push(() => {
      this.componentManager.addComponent(entity, component);
    });
    this.scheduleBatch();
  }

  private scheduleBatch(): void {
    if (this.batchTimer) return;
    
    this.batchTimer = setTimeout(() => {
      this.executeBatch();
    }, 0);
  }

  private executeBatch(): void {
    for (const operation of this.pendingOperations) {
      operation();
    }
    
    this.pendingOperations = [];
    this.batchTimer = null;
  }
}
```
